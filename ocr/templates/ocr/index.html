Quick rhythm

Choose a mode, add an image, extract the text, then listen or copy it. Everything on this page is built around that one flow.

Hands-free operation

Use voice commands or keyboard shortcuts for complete accessibility:

  • Say "scan" to open camera and capture
  • Say "read latest" to hear extracted text
  • Say "history" to access previous scans
  • Say "activate voice" to turn on voice commands
  • Press Alt+V to activate voice commands manually if needed
  • Press Alt+R to read aloud

/* ───────────────────────────────────────────── COMMAND PROCESSING ───────────────────────────────────────────── */ function processVoiceCommand(raw) { console.log('[CMD] raw:', raw); // ── normalise: strip punctuation, filler words const cmd = (' ' + raw .toLowerCase() .replace(/[^a-z0-9\s]/g, ' ') .replace(/\b(please|the|now|me|my|some|a|an|just)\b/g, ' ') .replace(/\s+/g, ' ') .trim() + ' '); console.log('[CMD] normalised:', cmd.trim()); // ── DEACTIVATE if (/\b(deactivate|disable|stop|turn off|pause)\s*(voice|listening|commands?)\b/.test(cmd) || /\bvoice\s*off\b/.test(cmd)) { voiceActive = false; wakeWordActive = false; clearRestartTimer(); if (isListening && recognition) { try { recognition.stop(); } catch(e){} } updateVoiceBtn(); announce('Voice commands deactivated. Say activate voice or press the mic button whenever you want me again.', true); return; } // ── HELP if (/\b(help|commands?|what can i say|what (commands?|can you do))\b/.test(cmd)) { const helpMsg = 'Available voice commands: ' + 'scan or take photo to open camera, ' + 'camera to open the camera, ' + 'read latest or read aloud to hear the extracted text, ' + 'read history to load saved scans, ' + 'history or recent scans to go to history, ' + 'deactivate voice to stop listening, ' + 'or say help again anytime.'; announce(helpMsg, true); return; } // ── READ HISTORY (must come before generic READ to avoid ambiguity) if (/\b(read|open|show|load|play)\s*(the\s*)?(history|recent|previous|saved|past)\b/.test(cmd)) { announce('Loading recent scans.', true); loadHistory(); const hist = document.getElementById('history'); if (hist) hist.scrollIntoView({ behavior: 'smooth' }); return; } // ── HISTORY (without read verb) if (/\b(history|recent\s*scans?|previous\s*scans?|saved\s*scans?|open\s*history|show\s*history)\b/.test(cmd)) { announce('Loading recent scans.', true); loadHistory(); const hist = document.getElementById('history'); if (hist) hist.scrollIntoView({ behavior: 'smooth' }); return; } // ── READ LATEST TEXT if (/\b(read|speak|say|play)\b/.test(cmd) && (/\b(latest|last|current|it|aloud|out\s*loud|text|result|extracted)\b/.test(cmd) || cmd.trim() === 'read' || cmd.trim() === 'speak')) { if (currentText && currentText.trim()) { readAloud(); } else { announce('No text to read yet. Please scan an image first.', false); } return; } // standalone "read" or "speak" with no other context if (/^\s*(read|speak)\s*$/.test(cmd)) { if (currentText && currentText.trim()) { readAloud(); } else { announce('No text to read yet. Please scan an image first.', false); } return; } // ── SCAN if (/\b(scan|take\s*photo|take\s*picture|take\s*a\s*photo|capture\s*(image)?|start\s*scan|do\s*a\s*scan)\b/.test(cmd)) { if (currentFile) { announce('Scanning image now.', true); submitScan(); } else { announce('Opening camera for scanning.', true); openCamera(); } return; } // ── CAMERA if (/\b(camera|open\s*camera|start\s*camera|use\s*camera)\b/.test(cmd)) { announce('Opening camera.', true); openCamera(); return; } // ── EXTRACT / RUN OCR (already have image) if (/\b(extract|run\s*(ocr|extraction)|process\s*(image|photo)?|analyse|analyze)\b/.test(cmd)) { if (currentFile) { announce('Extracting text now.', true); submitScan(); } else { announce('Please add an image first, then say extract.', false); } return; } // ── STOP READING if (/\b(stop|pause|quiet|silence|mute)\b/.test(cmd) && /\b(reading|speech|speaking|audio|voice|talking)\b/.test(cmd)) { if (speechSynthesis.speaking) { speechSynthesis.cancel(); const btn = document.getElementById('speakBtn'); if (btn) { btn.classList.remove('speaking'); btn.textContent = 'Read Aloud'; } showToast('Stopped reading.'); } return; } // ── NOT RECOGNISED console.log('[CMD] not recognised:', cmd.trim()); announce('I didn\'t catch that. You said: ' + raw + '. Say help for a list of commands.', true); } /* ───────────────────────────────────────────── TEXT-TO-SPEECH HELPERS ───────────────────────────────────────────── */ function announce(message, interrupt) { clearRestartTimer(); if (interrupt === undefined) interrupt = true; if (interrupt && speechSynthesis.speaking) { speechSynthesis.cancel(); } else if (!interrupt && speechSynthesis.speaking) { console.log('[TTS] busy – skipping:', message); return; } const utt = new SpeechSynthesisUtterance(message); utt.rate = 0.92; utt.pitch = 1.0; utt.volume = 1.0; utt.onend = function() { // After TTS: restart mic so user can give the next command immediately scheduleRestart(500); }; utt.onerror = function(ev) { console.warn('[TTS] error:', ev.error); scheduleRestart(600); }; try { speechSynthesis.speak(utt); } catch(e) { console.error('[TTS] speak failed:', e); scheduleRestart(600); } } // Toast notification for silent feedback when TTS can't run function showToast(message) { let toast = document.getElementById('va-toast'); if (!toast) { toast = document.createElement('div'); toast.id = 'va-toast'; toast.style.cssText = [ 'position:fixed', 'bottom:1.5rem', 'left:50%', 'transform:translateX(-50%)', 'background:#162231', 'color:#eaf1f7', 'border:1px solid rgba(56,201,153,0.35)', 'border-radius:999px', 'padding:0.65rem 1.4rem', 'font-size:0.95rem', 'box-shadow:0 12px 28px rgba(0,0,0,0.35)', 'z-index:9999', 'transition:opacity 0.3s ease', 'opacity:0', 'pointer-events:none' ].join(';'); document.body.appendChild(toast); } toast.textContent = message; toast.style.opacity = '1'; clearTimeout(toast._hideTimer); toast._hideTimer = setTimeout(function() { toast.style.opacity = '0'; }, 3500); }
